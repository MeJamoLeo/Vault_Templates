type RealEstateProperty = {
  // 基本情報
  "物件番号": string;
  "種別": string;
  "物件番号2"?: string;
  "物件番号3"?: string;
  "店舗コード": string;
  "物件名": string;
  "価格(万円)": number;
  "最終取引価格（万円）"?: number;

  // 所在地情報
  "所在地１": string;
  "所在地２": string;
  "所在番地": string;
  "小学校区": string;
  "中学校区": string;

  // 交通情報（配列で表現）
  "交通": Array<{
    "路線": string;
    "駅": string;
    "徒歩": number;
    "分譲地徒歩(から)"?: number;
    "分譲地徒歩(まで)"?: number;
  }>;

  // 土地情報
  "土地(m2)": number;
  "土地(坪)": number;
  "道路１"?: string;
  "道路２"?: string;
  "道路３"?: string;
  "私道負担面積(m2)"?: number;

  // 建物情報
  "建物(m2)"?: number;
  "建物(坪)"?: number;
  "建物構造（選択）"?: string;
  "間取り（部屋数）"?: number;
  "築年月：年"?: number;
  "築年月：月"?: number;

  // 管理情報
  "管理費・共益費(円)"?: number;
  "修繕積立金"?: number;
  "駐車場"?: {
    "台数": number;
    "駐車場月額"?: number;
  };

  // その他
  "特記事項"?: string[];
  "備考"?: string;
  "写真": string[];
  "登録日": Date;
  "変更日": Date;
};
